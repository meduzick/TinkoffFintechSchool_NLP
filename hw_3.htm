def tokenize(text, stopWords):
    vectors = []
    a = re.compile(r'(?<=[А-Яа-я0-9])\.(?=[А-Яа-я])')
    b = re.compile(r'(?<=[А-Яа-я])\.(?=[А-Яа-я0-9])')
    c = re.compile(r'(?<=[А-Яа-я])\.(?=[A-Za-z])')
    d = re.compile(r'(?<=[A-Za-z])\.(?=[А-Яа-я])')
    for line in text:
        line = a.sub(' ', line)
        line = b.sub(' ', line)
        line = c.sub(' ', line)
        line = d.sub(' ', line)
        vectors.append([i.strip('. -') for i in line.split() if i.strip('. -') not in stopWords \
                      and len(i.strip('. -')) > 1])
    return vectors

    # [batch_size x sentence_size x embedding_size]
    inputs = tf.contrib.layers.embed_sequence(
        features['x'], vocab_size, embedding_size,
        initializer=params['embeddings_initializer'])

    training = mode == tf.estimator.ModeKeys.TRAIN
    
    conv = tf.layers.conv1d(
        inputs=inputs,
        filters=32,
        kernel_size=3,
        padding="same",
        activation=tf.nn.relu)
    
    pool = tf.keras.layers.MaxPooling1D(pool_size=2, strides=1,data_format='channels_first')(conv)
    
    birnn = tf.keras.layers.Bidirectional(tf.keras.layers.SimpleRNN(10, return_sequences=True))(pool)
    
    birnnback = tf.keras.layers.Bidirectional(tf.keras.layers.SimpleRNN(10))(birnn)
    
    logits = tf.layers.dense(birnnback, units=3)
    

    # This will be None when predicting
    if labels is not None:

    optimizer = tf.train.AdamOptimizer()

    def _train_op_fn(loss):
        return optimizer.minimize(
            loss=loss,
            global_step=tf.train.get_global_step())

    return head.create_estimator_spec(
        features=features,
        labels=labels,
        mode=mode,
        logits=logits,
        train_op_fn=_train_op_fn)
